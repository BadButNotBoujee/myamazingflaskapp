
CREATE TABLE USERS (
	USERNAME VARCHAR(255) UNIQUE,
	PASSWORD VARCHAR(255),
	PRIMARY KEY (USERNAME)
);

CREATE TABLE REVIEW (
	REVIEW_ID SERIAL,
	USERNAME VARCHAR(255),
	
	COFFEE_SHOP VARCHAR(255),
	RATING INT,	
	TITLE VARCHAR(255),
	DESCRIPTION TEXT,
	
	PRIMARY KEY(REVIEW_ID),
	FOREIGN KEY(USERNAME) REFERENCES USERS(USERNAME)
);
